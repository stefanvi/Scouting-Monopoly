% Definitie van de kaartjes (inclusief parameters)
\newtcolorbox{kaartje}[2]{
    adjusted title=#1,         % de titel (parameter)
    halign title=center,       % centreer de titel
    coltitle=black,            % maak de titel zwart
    colbacktitle=#2,           % de kleur achter de titel (parameter)
    fonttitle=\Large\bfseries, % maak de titel groot en dikgedrukt
    fontupper=\Large\bfseries, % maak ook de prijs groot en dikgedrukt
    fontlower=\scriptsize,     % lettergrootte
    halign lower=center,       % centreer het deel onder de stippellijn
    halign upper=center        % centreer het deel boven de stippellijn
}
